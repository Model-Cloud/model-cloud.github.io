Install with one command

Download the latest release for your OS/arch and run it.

# Linux (amd64 example)
curl -L https://github.com/Model-Cloud/mc/releases/latest/download/mc-node-linux-amd64.tar.gz -o mc-node-linux-amd64.tar.gz \
  && tar -xzf mc-node-linux-amd64.tar.gz && ./mc-node-linux-amd64

# macOS (arm64 example)
curl -L https://github.com/Model-Cloud/mc/releases/latest/download/mc-node-darwin-arm64.tar.gz -o mc-node-darwin-arm64.tar.gz \
  && tar -xzf mc-node-darwin-arm64.tar.gz && ./mc-node-darwin-arm64

# Windows (PowerShell, amd64 example)
Invoke-WebRequest https://github.com/Model-Cloud/mc/releases/latest/download/mc-node-windows-amd64.zip -OutFile mc-node-windows-amd64.zip; \
  Expand-Archive mc-node-windows-amd64.zip; \
  .\mc-node-windows-amd64.exe

Manual install

  1. Download the archive for your OS and architecture.
  2. Extract the archive to a writable location.
  3. Set environment variables and run the binary.
# Linux/macOS
export MC_NODE_ID="your-node-id"
export MC_NODE_SECRET="your-node-secret"
export MC_GRPC_URL="localhost:9090"    # optional
export MC_MODELS_DIR="/tmp/mc-models"   # optional
# If llama-server is not in PATH
# export MC_LLAMA_SERVER_BIN_PATH="/path/to/llama-server"
./mc-node-<os>-<arch>
